CREATE TABLE Players (
    player_id SERIAL,
    "name" TEXT NOT NULL,
    surname TEXT NOT NULL,
    patronymic TEXT,
    date_of_birth DATE NOT NULL,
    passport_series INTEGER NOT NULL,
    passport_number INTEGER NOT NULL,
    division_number INTEGER NOT NULL,
    date_of_issue DATE NOT NULL,
    issued_by_whom TEXT NOT NULL,
    team_number INTEGER NOT NULL,
    height INTEGER NOT NULL,
    weight INTEGER NOT NULL,
    phone_number TEXT NOT NULL,
    salary INTEGER NOT NULL,
	nationality_id INT NOT NULL,
	position_id INT NOT NULL
);

ALTER TABLE Players 
ADD CONSTRAINT PK_Players 
    PRIMARY KEY (player_id);

ALTER TABLE Players 
ADD CONSTRAINT U_Players_phone_number 
    UNIQUE (phone_number);

ALTER TABLE Players 
ADD CONSTRAINT U_Players_team_number 
    UNIQUE (team_number);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_team_number 
    CHECK (team_number > -1);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_passport_number 
    CHECK (passport_number > 0);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_passport_series 
    CHECK (passport_series > 1850);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_division_number
    CHECK (division_number > 0);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_height
    CHECK (height > 1.5);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_weight
    CHECK (weight > 60);

ALTER TABLE Players 
ADD CONSTRAINT CH_Players_salary
    CHECK (salary > 0);
